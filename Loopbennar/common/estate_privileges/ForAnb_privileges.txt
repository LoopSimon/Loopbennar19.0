
estate_adventurers_bellacaire_sappers = {
	icon = privilege_artillery
	loyalty = 0
	influence = 0.1
	is_valid = {
		has_country_flag = guardsmen_unlocked
		is_or_was_tag = { tag = B59 } #Pashaine
	}
	
	can_select = {
		if = {
			limit = { NOT = { has_country_flag = guardsmen_privilege_no_limit } }
			NOT = {
				has_estate_privilege = estate_adventurers_napesbay_squadron
				has_estate_privilege = estate_adventurers_menibor_vanguard
				has_estate_privilege = estate_adventurers_bluehart_finest
			}
		}
		else = { always = yes }
	}

	can_revoke = { always = yes }

	penalties = {
		allowed_rev_guard_fraction = -0.1
	}

	benefits = {
		artillery_barrage_cost = -0.5
	}

	conditional_modifier = {
		trigger = {
			has_country_flag = guardsmen_privilege_no_limit
		}
		modifier = { land_morale = 0.025 }
	}
}

estate_burghers_committe_of_urban_development = {
	icon = privilege_development_efficiency
	max_absolutism = -10
	loyalty = 0.1
	influence = 0.1
	land_share = 0
	is_valid = {
		has_country_flag = pashaine_enable_urban_privilege
	}
	can_revoke = { always = yes }

	on_granted = {
		custom_tooltip = estate_burghers_committe_of_urban_development_grant_tt
		tooltip = {
			add_province_modifier = {
				name = pashaine_urban_development
				duration = -1
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_burghers_committe_of_urban_development_removed_tt
	}

	on_granted_province = {
		if = {
			limit = {
				OR = {
					has_terrain = city_terrain
					province_has_center_of_trade_of_level = 1
					has_river_estuary_trigger = yes
				}
			}
			add_province_modifier = {
				name = pashaine_urban_development
				duration = -1
			}
		}
	}

	on_invalid_province = {
		if = {
			limit = { has_province_modifier = pashaine_urban_development }
			remove_province_modifier = pashaine_urban_development
		}
	}

	on_revoked_province = {
		if = {
			limit = { has_province_modifier = pashaine_urban_development }
			remove_province_modifier = pashaine_urban_development
		}
	}
}
